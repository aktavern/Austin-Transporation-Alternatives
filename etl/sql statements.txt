CREATE TABLE all_vehicles(
	unique_id INT PRIMARY KEY,
  	date DATE,
	day INTEGER,
	month INTEGER,
	year INTEGER,
	hour INTEGER,
	day_of_week INTEGER,
	atemp INTEGER,
	ahum INTEGER,
	awind INTEGER,
	prec INTEGER,
	vehicle_encoded INTEGER,
	hour_encoded INTEGER,
	month_encoded INTEGER,
	count_encoded INTEGER,
	count INTEGER
	);


CREATE TABLE vehicles(
	vehicle_encoded INT PRIMARY KEY,
	vehicle_type VARCHAR(30)
	);

CREATE TABLE hours(
	hour_encoded INT PRIMARY KEY,
	hour_group VARCHAR(30)
	);

CREATE TABLE months(
	month_encoded INT PRIMARY KEY,
	month_group VARCHAR(30)
	);

CREATE TABLE counts(
	count_encoded INT PRIMARY KEY,
	count_group VARCHAR
	);

CREATE TABLE weather(
	date DATE PRIMARY KEY,
	maxtemp INTEGER,
	atemp INTEGER,
	mintemp INTEGER,
	maxhum INTEGER,
	ahum INTEGER,
	minhum INTEGER,
	maxwind INTEGER,
	awind INTEGER,
	minwind INTEGER,
	prec INTEGER,
	day INTEGER,
	month INTEGER,
	day_of_week INTEGER,
	year INTEGER
	);
